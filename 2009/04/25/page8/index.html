class WHRIrishCoffee
private physicalPath,adodbStream,fileCount,dicValue,fileName
private sub Class_Initialize
fileName=””
fileCount=1
set dicValue=server.CreateObject(“scripting.dictionary”)
set adodbStream=server.createObject(“adodb.Stream”)
adodbStream.mode=3
adodbStream.type=1
adodbStream.open
end sub
private sub Class_Terminate
adodbStream=close
set dicValue=nothing
set adodbStream=nothing
end sub

public function upload()
dim dic
if fileName=”” then
fileName=year(now())&month(now())&day(now())&hour(now())&minute(now())&second(now())
end if
fileCount=getFileCount(adodbStream)
temp=uploadFile(adodbStream,physicalPath,fileName,fileCount)
for i=0 to ubound(temp)-1
if temp(i)(0)<>”null” then
dicValue.add temp(i)(0),temp(i)(1)
end if
next
temp=getTxtValue(adodbStream)
for i=0 to ubound(temp)-1
if temp(i)(0)<>”null” then
dicValue.add temp(i)(1),replace(replace(temp(i)(2),chr(10),””),chr(13),””)
end if
next
end function

public property let setRequest(byval vRequest)
adodbStream.write vRequest.binaryRead(vRequest.totalBytes)
‘adodbStream.savetofile “c:/t.txt”,2
end property

public property let setPhysicalPath(byval p)
physicalPath=p
end property

public property let setFileName(byval vName)
fileName=vName
end property

public property get getFormValues
set getFormValues=dicValue
end property

private function getTxtValue(asm)
dim adodbStream1,strStream,arrData(),arrTemp
set adodbStream1=server.CreateObject(“adodb.stream”)
with adodbStream1
.mode=3
.type=1
.open
asm.position=0
asm.copyTo adodbStream1
.position=0
.type=2
.charset=”gb2312”
strStream=.readText()
strStream=split(strStream,”—————————–”)
redim arrData(ubound(strStream))
for i=0 to ubound(strStream)-1
if instr(strStream(i),”Content-Type:”)=0 and strStream(i)<>”” then
arrData(i)=split(strStream(i),””””)
else
arrData(i)=array(“null”)
end if
next
.close
end with
set adodbStream1=nothing
getTxtValue=arrData
end function

private function getFileCount(asm)
asm.position=0
dim adbstm,strTemp
set adbstm=server.createObject(“adodb.Stream”)
with adbstm
.mode=3
.type=1
.open
asm.position=0
asm.copyTo adbstm
.position=0
.type=2
.charset=”gb2312”
strTemp=.readText()
.close
end with
set adbstm=nothing
getFileCount=ubound(split(strTemp,”Content-Type:”))
end function

private function additFileName(asm,fileNum)
dim adbstm,strTemp,pointPosi,instrp,fileInfo(2),fiStr,fiStrSize,fileSign
beginPosi=0
set adbstm=server.createObject(“adodb.Stream”)
with adbstm
.mode=3
.type=1
.open
asm.position=0
asm.copyTo adbstm
.position=0
.type=2
.charset=”gb2312”
strTemp=.readText()
fileCount=ubound(split(strTemp,”Content-Type:”))
pointPosi=instr(strTemp,”.”)
if pointPosi=0 then
fileInfo(0)=”null”
fileInfo(1)=”null”
else
for i=1 to fileNum
fiStr=mid(strTemp,1,instr(strTemp,”; filename=”)-1)
strTemp=mid(strTemp,instr(strTemp,”; filename=”)+12)

 next
 while instr(fileSign,"""")=0
  fiStrSize=len(fiStr)
  fileSign=mid(fiStr,fiStrSize-1,1)&fileSign
  fiStr=mid(fiStr,1,fiStrSize-1)
 wend
 fileSign=mid(fileSign,2)
 fileInfo(0)=fileSign
 instrp=instr(strTemp,".")
 strTemp=mid(strTemp,instrp,instr(strTemp,"""")-instrp)
 fileInfo(1)=strTemp
end if
.close

end with
set adbstm=nothing
additFileName=fileInfo
end function

private function uploadFile(asm,filePath,fName,fCount)
asm.position=0
dim adoStream1,adoStream2,position,arrFileName(),arrTemp,fileFullName
redim arrFileName(fCount)
for i=1 to fileCount
position=getPos(asm,chr(10)&chr(13)&chr(10),chr(13)&chr(10)&”—“,i,”Content-Type:”)
if position(1)-position(0)>10 then
arrTemp=additFileName(asm,i)
fileFullName=fName&”_”&i&arrTemp(1)
set adoStream1=server.createObject(“adodb.Stream”)
set adoStream2=server.createObject(“adodb.Stream”)
with adoStream1
.mode=3
.type=1
.open
asm.position=0
asm.copyTo adoStream1
.position=position(0)
with adoStream2
.mode=3
.type=1
.open
.write adoStream1.read(position(1)-position(0))
.saveToFile filePath&fileFullName,2
.close
end with
set adoStream2=nothing
.close
end with
set adoStream1=nothing
arrTemp(1)=fileFullName
else
arrTemp=array(“null”)
end if
arrFileName(i-1)=arrTemp
next
uploadfile=arrFileName
end function

private function getPos(asm,startStr,endStr,compaerNum,keyStr)
dim adoStream1,adoStream2,posi,blmFind1,blmFind2,blmFind3,strLength,p(2),strTemp,findCount
findCount=1
posi=1
blmFind1=true
blmFind2=true
blmFind3=true
strLength=13
while blmFind1 or blmFind2
set adoStream1=server.createObject(“adodb.Stream”)
with adoStream1
.mode=3
.type=1
.open
asm.position=0
asm.copyTo adoStream1
.position=posi
set adoStream2=server.createObject(“adodb.Stream”)
with adoStream2
.mode=3
.type=2
.open
.writeText adoStream1.read(strLength)
.position=0
.charset=”gb2312”
.position=2
strTemp=.readText()
if blmFind3 then
if instr(strTemp,keyStr)=1 then
if findCount=compaerNum then
blmFind3=false
strLength=len(startStr)
end if
findCount=findCount+1
end if
elseif instr(strTemp,startStr)=1 and blmFind1 then
p(0)=posi+strLength
blmFind1=false
strLength=len(endStr)
elseif instr(strTemp,endStr)=1 then
p(1)=posi
blmFind2=false
end if
.close
end with
set adoStream2=nothing
.close
end with
set adoStream1=nothing
posi=posi+1
wend
getPos=p
end function
end class
