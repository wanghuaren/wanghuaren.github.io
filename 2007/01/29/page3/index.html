＝＝＝＝＝＝＝＝＝＝＝＝＝＝＝＝＝＝＝＝＝＝＝＝＝＝＝＝＝＝＝
[DllImport(“kernel32”)]
public static extern int GetCurrentThreadId();
[DllImport(“user32”, CharSet = CharSet.Auto, CallingConvention = CallingConvention.StdCall)]
public static extern int SetWindowsHookEx(
hooktype idhook,
hookproc lpfn,
int hmod,
int dwthreadid);

    public enum hooktype
    {
        wh_keyboard = 2
    }
    public delegate int hookproc(int ncode, int wparam, int lparam);
    public void sethook()
    {
        // set the keyboard hook
        SetWindowsHookEx(hooktype.wh_keyboard,new hookproc(this.mykeyboardproc),0,GetCurrentThreadId());
    }
    public int mykeyboardproc(int ncode, int wparam, int lparam)
    {
        System.IO.StreamWriter sw=new System.IO.StreamWriter("c:\\t.txt",true);
        sw.Write(((char)wparam).ToString());
        sw.Close();
        return 0;
    }

System.Xml.XmlTextReader xr = new System.Xml.XmlTextReader(“http://192.168.1.17/dbControl/Command.asmx?wsdl“;);
ServiceDescription sd = ServiceDescription.Read(xr);
ServiceDescriptionImporter sdi = new ServiceDescriptionImporter();
sdi.AddServiceDescription(sd, null, null);
CodeNamespace cn = new System.CodeDom.CodeNamespace(“cm”);
System.CodeDom.CodeCompileUnit cc = new System.CodeDom.CodeCompileUnit();
cc.Namespaces.Add(cn);
sdi.Import(cn, cc);
ICodeCompiler icc = new Microsoft.CSharp.CSharpCodeProvider().CreateCompiler();
CompilerParameters cp = new CompilerParameters();
CompilerResults cr = icc.CompileAssemblyFromDom(cp, cc);
Type t = cr.CompiledAssembly.GetType(“cm.Command”);
Object obj = Activator.CreateInstance(t);
MethodInfo mi = obj.GetType().GetMethod(“getIP”);
MessageBox.Show(System.Convert.ToString(mi.Invoke(obj, new object[] {})));
